var message = 'Hello World!!';
var span = document.createElement('span');
document.body.appendChild(span);

var promise5sec = new Promise(function (resolve) {
	setTimeout(function () {
		span.innerHTML = message;
		resolve();
	}, 5000);
});

function swap(x, y) {
	var _x = x;
	x = y;
	y = _x;
	return [x, y];
}

var x = 5;
var y = 6;
var swapped = swap(x, y);
span.innerHTML = '[' + swapped[0] + ', ' + swapped[1] + ']';
